作者：郑世贤
版本：v 1.0
邮箱：solvemyquestion@163.com
日期：2011-12-25

使用方法：

核心样式文件：scrollH.css

核心JS文件：jquery.js库 jquery.scrollH.js

插件引用：

		/*默认版*/
		$('.scrollH').scrollH();

		/*完全版 - 手动*/
		$('.scrollH').scrollH({
			type : 'Manual',			   /*滚动的方式: Manual(手动),Auto(自动)*/
			line : 1,					   /*每次滚动的个数*/
			time : 800,					   /*滚动的时间 单位:毫秒*/
			btnUp : 'btn-left',			   /*触发向上滚动的按钮的class名*/
			btnDown : 'btn-right'		   /*触发向下滚动的按钮的class名*/
		});
        
        /*完全版 - 自动*/
		$('.scrollH').scrollH({
			type : 'Auto',			       /*滚动的方式: Manual(手动),Auto(自动)*/
			line : 1,					   /*每次滚动的个数*/
			time : 800,					   /*滚动的时间 单位:毫秒*/
			blankTime : 3000,			   /*自动滚动的间隔时间 单位:毫秒*/
			direction : 'Left'			   /*自动滚动的方向 此为可选参数 Left(向左边-默认),Right(向右)*/
		});
